9   public class Kenttamestari {
10  
11      private Scanner lukija;
12      private File tiedosto;
13      private List<Kentta> kenttaLista;
14  
15      public Kenttamestari() {
16          lukija = null;
17          tiedosto = new File("src/main/resources/pohjat.txt");
18          kenttaLista = new ArrayList<>();
19      }
20  
21      public Scanner haeTiedosto() {
22          try {
23              lukija = new Scanner(tiedosto, "UTF-8");    // This will be closed in haeKaikkiKentatListalle() method
24          } catch (Exception e) {
25              System.out.println("Tiedoston lukeminen ep√§onnistui. Virhe: " + e.getMessage());
26          }
27          return lukija;
28      }
29  
30      public void haeKaikkiKentatListalle() {
31          while (lukija.hasNextLine()) {
32              String rivi = lukija.nextLine();
33              String[] osat = rivi.split(" ");                                // Read each line from /src/main/resources/pohjat.txt
34              int vaikeustaso = Integer.valueOf(osat[0]);                     // and split into diff. level, player view and solution
35              kenttaLista.add(new Kentta(vaikeustaso, osat[1], osat[2]));     // make a list of these new bases
36          }
37          lukija.close();                                                     // Close the scanner opened in haeTiedosto
38      }
39      
40      public Kentta annaKentta() {
41          Collections.shuffle(kenttaLista);       // Randomize the items on the list
42          return kenttaLista.get(0);              // and return one
43      }
