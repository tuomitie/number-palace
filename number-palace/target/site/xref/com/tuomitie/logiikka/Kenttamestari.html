1   package com.tuomitie.logiikka;

3   import java.io.File;
4   import java.util.ArrayList;
5   import java.util.Collections;
6   import java.util.List;
7   import java.util.Scanner;
8   
9   /**
10   * Reads Kentta (field) classes from a file and serves a random one to be used
11   * as a basis for the game to be played.
12   *
13   * @author Tuomas
14   */
15  public class Kenttamestari {
16  
17      private Scanner lukija;
18      private File tiedosto;
19      private List<Kentta> kenttaLista;
20  
21      public Kenttamestari() {
22          lukija = null;
23          tiedosto = new File("src/main/resources/pohjat.txt");
24          kenttaLista = new ArrayList<>();
25      }
26  
27      /**
28       * Tries to read the file specified in the constructor.
29       *
30       * @return
31       */
32      public Scanner haeTiedosto() {
33          try {
34              lukija = new Scanner(tiedosto, "UTF-8");    // This will be closed in haeKaikkiKentatListalle() method
35          } catch (Exception e) {
36              System.out.println("Tiedoston lukeminen ep√§onnistui. Virhe: " + e.getMessage());
37          }
38          return lukija;
39      }
40  
41      /**
42       * Adds all the read lines on a list, each as an numeric difficulty level
43       * and two strings in an array.
44       */
45      public void haeKaikkiKentatListalle() {
46          while (lukija.hasNextLine()) {
47              String rivi = lukija.nextLine();
48              String[] osat = rivi.split(" ");                                // Read each line from /src/main/resources/pohjat.txt
49              int vaikeustaso = Integer.valueOf(osat[0]);                     // and split into diff. level, player view and solution
50              kenttaLista.add(new Kentta(vaikeustaso, osat[1], osat[2]));     // make a list of these new bases
51          }
52          lukija.close();                                                     // Close the scanner opened in haeTiedosto
53      }
54  
55      /**
56       * Returns a random Kentta from the list.
57       */
58      public Kentta annaKentta() {
59          Collections.shuffle(kenttaLista);       // Randomize the items on the list
60          return kenttaLista.get(0);              // and return one
61      }
62  }
