3   /**
4    * Kentta is the representation of a Sudoku board in its initial state.
5    *
6    * @author Tuomas
7    */
8   public class Kentta implements Comparable<Kentta> {
9   
10      private int vaikeustaso;
11      private String nakyma;
12      private String vastaus;
13  
14      /**
15       * The constructor must be called with all of the three variables of a
16       * Kentta.
17       *
18       * @param vaikeustaso The difficulty level.
19       * @param nakyma The player view with hidden cells.
20       * @param vastaus The correct solution to the puzzle.
21       */
22      public Kentta(int vaikeustaso, String nakyma, String vastaus) {
23          this.vaikeustaso = vaikeustaso;
24          this.nakyma = nakyma;
25          this.vastaus = vastaus;
26      }
27  
28      /**
29       * Used to convert the Strings read from file to an array of integers.
30       *
31       * @param data The string with a representation of the board.
32       * @return The one dimensional array of integers made from the input string.
33       */
34      public int[] taulukkoNumeroina(String data) {
35          if (data.contains(".")) {                       // In the case of a player view get rid of the dots
36              data = data.replaceAll("\\.", "0");         // and replace them with zeros
37          }
38          String[] merkit = data.split("");               // Split the String into pieces
39          int[] numerot = new int[81];
40          for (int i = 0; i < merkit.length; i++) {
41              numerot[i] = Integer.valueOf(merkit[i]);    // Convert from String to int
42          }
43          return numerot;
44      }
45  
46      public int haeVaikeustaso() {
47          return vaikeustaso;
48      }
49  
50      public String haeVastaus() {
51          return vastaus;
52      }
53  
54      public String haeNakyma() {
55          return nakyma;
56      }
57  
58      @Override
59      public int compareTo(Kentta verrattava) {           // For comparing games by based on difficulty
60          return this.vaikeustaso - verrattava.haeVaikeustaso();
61      }
62  }
