15      public void alusta() {
16          int[] osat = luoPohja();
17          /* Fill the 9x9 grid with the preset data from haePohja() method */
18          int indeksi = 0;                        // Index used in going through the source int array
19          for (int a = 0; a < 9; a++) {           // Rows of the array 
20              for (int b = 0; b < 9; b++) {       // Cells of a row
21                  rivit[a][b] = osat[indeksi];
22                  indeksi++;
23              }
24          }
25          luoPelaajanNakyma();
26      }

67      public void luoPelaajanNakyma() {
68          /* Creates the player view with hidden cells */
69          String peitto = ". 5 . . . 3 . . . . . 4 . . 8 . . 6 . . . . 5 2 . 4 . 9 7 3 . . . . . . . . 6 . . . 3 . . . . . . . . 6 9 7 . 1 . 9 2 . . . . 5 . . 1 . . 9 . . . . . 5 . . . 3 .";
70          String[] osat = peitto.split(" ");
71          int indeksi = 0;
72  
73          for (int a = 0; a < 9; a++) {
74              for (int b = 0; b < 9; b++) {
75                  if (osat[indeksi].equals(".")) {    // If the cell in question contains a dot, hide it
76                      peliKentta[a][b] = 0;
77                  } else {
78                      peliKentta[a][b] = rivit[a][b]; // If the cell contains a number, display it
79                  }
80                  indeksi++;
81              }
82          }
83      }

85      public void asetaNumero(int a, int b, int x) {
86          a--;
87          b--;
88          if ((a < 9) & (a >= 0) & (b < 9) & (b >= 0) & (x <= 9) & (x > 0)) {
89              if (peliKentta[a][b] == 0) {
90                  peliKentta[a][b] = x;
91              }
92          }
93      }

95      public int haePelikentanSolu(int a, int b) {
96          if ((a < 9) & (a >= 0) & (b < 9) & (b >= 0)) {
97              return peliKentta[a][b];
98          } else {
99              return -1;
100         }
