1   package com.tuomitie.gui;
2   
3   import com.tuomitie.logiikka.Sudoku;
4   import java.awt.event.ActionEvent;
5   import java.awt.event.ActionListener;
6   import javax.swing.JButton;
7   
8   public class KlikkauksenKuuntelija implements ActionListener {
9   
10      private final int x;
11      private final int y;
12      private final JButton button;
13      private Sudoku sudoku;
14  
15      /**
16       * KlikkauksenKuuntelija handles the mouse clicks on the player board.
17       * 
18       * @param x The row of the location of the button.
19       * @param y The column of the location.
20       * @param button The button in question.
21       * @param sudoku The Sudoku game the button affects.
22       */
23      public KlikkauksenKuuntelija(int x, int y, JButton button, Sudoku sudoku) {
24          this.x = x;
25          this.y = y;
26          this.button = button;
27          this.sudoku = sudoku;
28      }
29  
30      /**
31       * Increments the cell in the game and the button representing it on the board.
32       * Only displays the valid Sudoku numbers 1-9, and resets to (hidden) zero after reaching 9.
33       */
34      public void kasvata() {
35          int luku = sudoku.haePelikentanSolu(x, y);
36          if (luku >= 0 & luku < 9) {
37              luku++;
38              sudoku.asetaNumero("" + (x + 1) + (y + 1) + luku);
39              button.setText(String.valueOf(luku));
40          } else if (luku == 9) {
41              sudoku.asetaNumero("" + (x + 1) + (y + 1) + 0);
42              button.setText(" ");
43          }
44      }
45  
46      @Override
47      public void actionPerformed(ActionEvent ae) {
48          kasvata();
49      }
50  }
