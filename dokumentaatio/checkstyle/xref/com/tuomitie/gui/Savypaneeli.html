15  /**
16   * A custom JPanel that has a color gradient running across it.
17   *
18   * @author tuomas
19   */
20  public class Savypaneeli extends JPanel {
21  
22      private Color alkuvari;
23      private Color loppuvari;
24      private BufferedImage logo;
25  
26      /**
27       * Constructor takes the two colors used in the gradient.
28       *
29       * @param alkuvari Start color.
30       * @param loppuvari End color.
31       */
32      public Savypaneeli(Color alkuvari, Color loppuvari) {
33          super();
34          this.alkuvari = alkuvari;
35          this.loppuvari = loppuvari;
36          try {
37              InputStream is = getClass().getClassLoader().getResourceAsStream("number-palace-header.png");
38              logo = ImageIO.read(is);
39          } catch (IOException ex) {
40              System.out.println("Logoa ei l√∂ytynyt");
41          }
42      }
43  
44      @Override
45      protected void paintComponent(Graphics g) {
46          super.paintComponent(g);
47          int korkeus = getHeight();
48          int leveys = getWidth();
49          GradientPaint gradientPaint = new GradientPaint(0, 0, alkuvari, leveys, korkeus, loppuvari);
50          if (g instanceof Graphics2D) {
51              Graphics2D graphics2D = (Graphics2D) g;
52              graphics2D.setPaint(gradientPaint);
53              graphics2D.fillRect(0, 0, leveys, korkeus);
54              if (korkeus > 200 && logo != null) {
55                  g.drawImage(logo, 90, 20, this); // see javadoc for more info on the parameters
56              }
57          }
58      }

62      }
