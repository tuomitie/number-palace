18  public class Kayttoliittyma implements Runnable {
19  
20      private int[][] tilanne;
21      private JFrame frame;
22      private JButton[][] alkiot;
23      private JPanel pelialue;
24      private JPanel[][] blokit;
25      private Color reunat;
26      private Font numeroFontti;
27  
28      public Kayttoliittyma(int[][] tilanne) {
29          alkiot = new JButton[9][9];
30          pelialue = new JPanel(new GridLayout(3, 3));
31          blokit = new JPanel[3][3];
32          this.tilanne = tilanne;
33          numeroFontti = new Font("Sans-Serif", Font.BOLD, 21);
34      }
35  
36      @Override
37      public void run() {
38          frame = new JFrame("number-palace");
39          frame.setPreferredSize(new Dimension(500, 410));
40          frame.setLocation(300, 200);
41          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
42  
43          luoKomponentit(frame.getContentPane());
44  
45          frame.pack();
46          frame.setVisible(true);
47      }
48  
49      private void luoKomponentit(Container container) {
50          container.setLayout(new FlowLayout());
51  
52          pelialue.setPreferredSize(new Dimension(360, 360));     // The game board
53          Border line = new LineBorder(Color.GRAY);
54          Border margin = new EmptyBorder(10, 10, 10, 10);
55          Border compound = new CompoundBorder(margin, line);
56          pelialue.setBorder(compound);
57  
58          JPanel painikkeet = new JPanel(new FlowLayout());       // Control button panel
59          JButton keskeyta = new JButton("Keskeyt√§");
60          painikkeet.add(keskeyta);
61  
62          container.add(pelialue);
63          container.add(painikkeet);
64  
65          luoBlokit();                                            // Create the 3x3 sub-blocks for the board
66          alustaAlkiot();                                         // Create the number cells
67      }
68  
69      public void luoBlokit() {               // Create the smaller 3x3 blocks (mostly for visual purposes here)
70          for (int x = 0; x < 3; x++) {
71              for (int y = 0; y < 3; y++) {
72                  JPanel blokki = new JPanel(new GridLayout(3, 3, 0, 0));    // 3x3 blocks with zero margin
73                  Border viiva = new LineBorder(Color.BLACK);
74                  blokki.setBorder(viiva);
75                  blokki.setBackground(Color.white);
76                  blokit[x][y] = blokki;
77                  pelialue.add(blokki);
78              }
79          }
80      }
81  
82      public void alustaAlkiot() {
83          for (int a = 0; a < 9; a++) {               // Rows
84              for (int b = 0; b < 9; b++) {           // Cells
85                  JButton button = new JButton(" ");  // Init with empty string
86                  if (tilanne[a][b] != 0) {           // If contains a "given" set a number and make disabled
87                      button.setText("" + tilanne[a][b]);
88                      button.setEnabled(false);
89                  } else {                            // If empty, make clickable
90                      button.addActionListener(new KlikkauksenKuuntelija(a, b));
91                  }
92                  button.setContentAreaFilled(false); // Hide cell styling
93                  button.setPreferredSize(new Dimension(30, 30));
94                  button.setFont(numeroFontti);
95                  button.setBorder(new LineBorder(Color.lightGray));
96                  alkiot[a][b] = button;              // Add into the 9x9 button table
97                  blokit[(int) (a / 3)][(int) (b / 3)].add(button);   // Add into block [0, 1 or 2][0, 1 or 2]
98              }
99          }
100     }
101 
102     public JFrame getFrame() {
103         return frame;
104     }
105 }
