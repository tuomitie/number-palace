41      public void suorita() {
42          if (toiminto.equals("vastaus")) {
43              kayttoliittyma.naytaRatkaisu();
44              kayttoliittyma.disabloi("tarkista");
45              kayttoliittyma.disabloi("katso vastaus");
46          } else if (toiminto.equals("tarkista")) {
47              if (sudoku.tarkistaRatkaisu()) {
48                  kayttoliittyma.luoPopUp("Oikein!!!");
49              } else {
50                  kayttoliittyma.luoPopUp("Väärin meni.");
51              }
52          } else if (toiminto.equals("aloita")) {
53              // TODO Implement start screen functionality
54              System.out.println("Aloita painettu.");
55          }
56      }
