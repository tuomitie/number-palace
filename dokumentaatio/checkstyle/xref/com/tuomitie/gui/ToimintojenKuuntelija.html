1   package com.tuomitie.gui;
2   
3   import com.tuomitie.logiikka.Sudoku;
4   import java.awt.event.ActionEvent;
5   import java.awt.event.ActionListener;
6   
7   public class ToimintojenKuuntelija implements ActionListener {
8   
9       private String toiminto;
10      private Kayttoliittyma kali;
11      private Sudoku sudoku;
12  
13      /**
14       * ActionListener for the user control buttons of the GUI.
15       * @param toiminto The action in question.
16       * @param kali The GUI instance.
17       * @param sudoku The Sudoku game the action affects.
18       */
19      public ToimintojenKuuntelija(String toiminto, Kayttoliittyma kali, Sudoku sudoku) {
20          this.toiminto = toiminto;
21          this.kali = kali;
22          this.sudoku = sudoku;
23      }
24  
25      /**
26       * Handles the button specific actions that should happen when clicked.
27       */
28      public void suorita() {
29          if (toiminto.equals("vastaus")) {
30              kali.naytaRatkaisu();
31              kali.disabloi("tarkista");
32              kali.disabloi("katso vastaus");
33          } else if (toiminto.equals("tarkista")) {
34              if (sudoku.tarkistaRatkaisu()) {
35                  kali.luoPopUp("Oikein!!!");
36              } else {
37                  kali.luoPopUp("Väärin meni.");
38              }
39          }
40      }
41  
42      @Override
43      public void actionPerformed(ActionEvent ae) {
44          suorita();
45      }
46  }
